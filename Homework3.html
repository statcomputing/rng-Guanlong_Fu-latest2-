mixbeta <- function(x,n,a,b,lambda){


u2 <- runif(n, min = 0, max = 1)  # sample u2
v <- runif(n, min = 0, max = 1)   # sample v

    for (i in 1:length(u2)) {
      if (v[i]> lambda) {
        x[i]=u2[i]^(1/a)
     } else {
       x[i]=1-u2[i]^(1/b)
     }
    }
 return(x)
}
